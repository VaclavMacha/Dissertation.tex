% ------------------------------------------------------------------------------
% Tikz + PGFPlots styles
% ------------------------------------------------------------------------------
\pgfplotsset{
  lineiver/.style  = {smooth, myred, very thick},
  line/.style  = {no markers, smooth, very thick},
  linequadratic/.style  = {no markers, smooth, mygreen, very thick},
}

\pgfmathdeclarefunction{iverson}{1}{%
  \pgfmathparse{max(0, 1 + #1)}%
}
\pgfmathdeclarefunction{hinge}{2}{%
  \pgfmathparse{max(0, 1 + #2 * #1)}%
}
\pgfmathdeclarefunction{quadratic}{2}{%
  \pgfmathparse{(max(0, 1 + #2 * #1))^2}%
}

\begin{document}
%
\begin{tikzpicture}
  \begin{axis}[
    xmin = -2,
    xmax = 2.1,
    ymin = 0,
    ymax = 4.2,
    domain = -2:2,
    axis lines = left,
    xtick = {-2, -1, 0, 1, 2},
    xlabel={$s$},
    ylabel={$l(s)$},
    grid=both,
    grid style=dashed,
    major grid style={line width=.2pt,draw=gray!50},
    legend pos =north west,
    legend cell align={left},
    enlargelimits = false,
    height=6cm,
    width=14cm,
  ]
    
    \addplot [line, cyan] {hinge(x, 1)};
    \addlegendentry{Hinge loss~$\vartheta = 1$}
    \addplot [line, myorange, dashed] {hinge(x, 2)};
    \addlegendentry{Hinge loss~$\vartheta = 2$}
    \addplot [line, myblue, dotted] {quadratic(x, 1)};
    \addlegendentry{Quadratic hinge loss~$\vartheta = 1$}
    \addplot [line, mygreen, dashdotted] {quadratic(x, 2)};
    \addlegendentry{Quadratic hinge loss~$\vartheta = 2$}
    \addplot[lineiver, no markers] coordinates {(-2, 0) (0, 0)};
    \addlegendentry{Iverson funcion}
    \addplot[lineiver, no markers] coordinates {(0, 1) (2, 1)};
    \addplot[lineiver, only marks, mark=*, mark options={fill=white}] coordinates {(0, 0)};
    \addplot[lineiver, only marks, mark=*, mark options={fill=myred}] coordinates {(0, 1)};
  \end{axis}
\end{tikzpicture}
\end{document}
